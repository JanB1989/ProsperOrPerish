# Prosper or Perish - Game Start Building Setup
# Adds cookeries to towns and cities based on population and development thresholds.
on_game_start = {
	on_actions = {
		pp_precalculate_building_capacities
		pp_remove_invalid_buildings
		pp_game_start_effect
	}
}
pp_precalculate_building_capacities = {
	effect = {
		every_location_in_the_world = {
			set_variable = { name = pp_fruit_orchard_fixed_env_bonus value = 0 }
			set_variable = { name = pp_fishing_village_fixed_env_bonus value = 0 }
			set_variable = { name = pp_sheep_farms_fixed_env_bonus value = 0 }
			set_variable = { name = pp_farming_village_fixed_env_bonus value = 0 }
			set_variable = { name = pp_forest_village_fixed_env_bonus value = 0 }
		}
		every_location_in_the_world = {
			limit = { is_land = yes }
			set_variable = {
				name = pp_fruit_orchard_fixed_env_bonus
				value = pp_fruit_orchard_capacity_value
			}
			set_variable = {
				name = pp_fishing_village_fixed_env_bonus
				value = pp_fishing_village_capacity_value
			}
			set_variable = {
				name = pp_sheep_farms_fixed_env_bonus
				value = pp_sheep_farms_capacity_value
			}
			set_variable = {
				name = pp_farming_village_fixed_env_bonus
				value = pp_farming_village_capacity_value
			}
			set_variable = {
				name = pp_forest_village_fixed_env_bonus
				value = pp_forest_village_capacity_value
			}
		}
		# Initialize Global Min/Max
		set_global_variable = { name = pp_fruit_orchard_global_min value = 100 }
		set_global_variable = { name = pp_fruit_orchard_global_max value = 0 }
		set_global_variable = { name = pp_fishing_village_global_min value = 100 }
		set_global_variable = { name = pp_fishing_village_global_max value = 0 }
		set_global_variable = { name = pp_sheep_farms_global_min value = 100 }
		set_global_variable = { name = pp_sheep_farms_global_max value = 0 }
		set_global_variable = { name = pp_farming_village_global_min value = 100 }
		set_global_variable = { name = pp_farming_village_global_max value = 0 }
		set_global_variable = { name = pp_forest_village_global_min value = 100 }
		set_global_variable = { name = pp_forest_village_global_max value = 0 }

		every_location_in_the_world = {
			limit = { is_land = yes }
			# Fruit Orchard
			if = {
				limit = { var:pp_fruit_orchard_fixed_env_bonus > global_var:pp_fruit_orchard_global_max }
				set_global_variable = { name = pp_fruit_orchard_global_max value = var:pp_fruit_orchard_fixed_env_bonus }
			}
			if = {
				limit = { var:pp_fruit_orchard_fixed_env_bonus < global_var:pp_fruit_orchard_global_min }
				set_global_variable = { name = pp_fruit_orchard_global_min value = var:pp_fruit_orchard_fixed_env_bonus }
			}
			# Fishing Village
			if = {
				limit = { var:pp_fishing_village_fixed_env_bonus > global_var:pp_fishing_village_global_max }
				set_global_variable = { name = pp_fishing_village_global_max value = var:pp_fishing_village_fixed_env_bonus }
			}
			if = {
				limit = { var:pp_fishing_village_fixed_env_bonus < global_var:pp_fishing_village_global_min }
				set_global_variable = { name = pp_fishing_village_global_min value = var:pp_fishing_village_fixed_env_bonus }
			}
			# Sheep Farms
			if = {
				limit = { var:pp_sheep_farms_fixed_env_bonus > global_var:pp_sheep_farms_global_max }
				set_global_variable = { name = pp_sheep_farms_global_max value = var:pp_sheep_farms_fixed_env_bonus }
			}
			if = {
				limit = { var:pp_sheep_farms_fixed_env_bonus < global_var:pp_sheep_farms_global_min }
				set_global_variable = { name = pp_sheep_farms_global_min value = var:pp_sheep_farms_fixed_env_bonus }
			}
			# Farming Village
			if = {
				limit = { var:pp_farming_village_fixed_env_bonus > global_var:pp_farming_village_global_max }
				set_global_variable = { name = pp_farming_village_global_max value = var:pp_farming_village_fixed_env_bonus }
			}
			if = {
				limit = { var:pp_farming_village_fixed_env_bonus < global_var:pp_farming_village_global_min }
				set_global_variable = { name = pp_farming_village_global_min value = var:pp_farming_village_fixed_env_bonus }
			}
			# Forest Village
			if = {
				limit = { var:pp_forest_village_fixed_env_bonus > global_var:pp_forest_village_global_max }
				set_global_variable = { name = pp_forest_village_global_max value = var:pp_forest_village_fixed_env_bonus }
			}
			if = {
				limit = { var:pp_forest_village_fixed_env_bonus < global_var:pp_forest_village_global_min }
				set_global_variable = { name = pp_forest_village_global_min value = var:pp_forest_village_fixed_env_bonus }
			}
		}
		
		# Calculate Ranges
		set_global_variable = {
			name = pp_fruit_orchard_global_range
			value = global_var:pp_fruit_orchard_global_max
		}
		change_global_variable = {
			name = pp_fruit_orchard_global_range
			add = { value = global_var:pp_fruit_orchard_global_min multiply = -1 }
		}
		
		set_global_variable = {
			name = pp_fishing_village_global_range
			value = global_var:pp_fishing_village_global_max
		}
		change_global_variable = {
			name = pp_fishing_village_global_range
			add = { value = global_var:pp_fishing_village_global_min multiply = -1 }
		}

		set_global_variable = {
			name = pp_sheep_farms_global_range
			value = global_var:pp_sheep_farms_global_max
		}
		change_global_variable = {
			name = pp_sheep_farms_global_range
			add = { value = global_var:pp_sheep_farms_global_min multiply = -1 }
		}

		set_global_variable = {
			name = pp_farming_village_global_range
			value = global_var:pp_farming_village_global_max
		}
		change_global_variable = {
			name = pp_farming_village_global_range
			add = { value = global_var:pp_farming_village_global_min multiply = -1 }
		}

		set_global_variable = {
			name = pp_forest_village_global_range
			value = global_var:pp_forest_village_global_max
		}
		change_global_variable = {
			name = pp_forest_village_global_range
			add = { value = global_var:pp_forest_village_global_min multiply = -1 }
		}
	}
}

# Remove all buildings that cannot exist under mod rules (Option B).
# Runs after precalculate so pp_* variables are set for fruit_orchard_max_level etc.
pp_remove_invalid_buildings = {
	effect = {
		every_location_in_the_world = {
			# Granary: only allowed in province capitals
			if = {
				limit = {
					NOT = { is_province_capital = yes }
					location_building_level = {
						building_type = building_type:granary
						value > 0
					}
				}
				change_building_level_in_location = {
					building = building_type:granary
					value = {
						value = "location_building_level(building_type:granary)"
						multiply = -1
					}
				}
			}
			# Fruit orchard: reduce to max (script value floored at 0)
			if = {
				limit = {
					location_building_level = {
						building_type = building_type:fruit_orchard
						value > 0
					}
					location_building_level = {
						building_type = building_type:fruit_orchard
						value > fruit_orchard_max_level
					}
				}
				change_building_level_in_location = {
					building = building_type:fruit_orchard
					value = {
						value = fruit_orchard_max_level
						add = {
							value = "location_building_level(building_type:fruit_orchard)"
							multiply = -1
						}
					}
				}
			}
			# Market village: reduce to rural_building_cap
			if = {
				limit = {
					location_building_level = {
						building_type = building_type:market_village
						value > 0
					}
					location_building_level = {
						building_type = building_type:market_village
						value > rural_building_cap
					}
				}
				change_building_level_in_location = {
					building = building_type:market_village
					value = {
						value = rural_building_cap
						add = {
							value = "location_building_level(building_type:market_village)"
							multiply = -1
						}
					}
				}
			}
		}
	}
}

pp_game_start_effect = {
	effect = {
		every_country = {
			limit = { exists = capital }
			capital ?= {
				if = {
					limit = { has_owner = yes }
					construct_building = {
						building_type = building_type:cookery
						instant = yes
						cost_multiplier = 0
						cost_multiplier_reason = "game_start"
					}
					construct_building = {
						building_type = building_type:victuals_market
						instant = yes
						cost_multiplier = 0
						cost_multiplier_reason = "game_start"
					}
				}
			}
			every_owned_location = {
				limit = { has_owner = yes }
				if = {
					limit = {
						location_rank = location_rank:rural_settlement
						population > 500
						NOT = { has_building = building_type:cookery }
					}
					construct_building = {
						building_type = building_type:cookery
						instant = yes
						cost_multiplier = 0
						cost_multiplier_reason = "game_start"
					}
				}

				if = {
					limit = {
						location_rank = location_rank:town
						population > 75
						development > 25
						NOT = { has_building = building_type:cookery }
					}
					construct_building = {
						building_type = building_type:cookery
						instant = yes
						cost_multiplier = 0
						cost_multiplier_reason = "game_start"
					}
				}
				else_if = {
					limit = {
						location_rank = location_rank:city
						population > 150
						development > 30
					}
					if = {
						limit = { NOT = { has_building = building_type:cookery } }
						construct_building = {
							building_type = building_type:cookery
							instant = yes
							cost_multiplier = 0
							cost_multiplier_reason = "game_start"
						}
					}
					construct_building = {
						building_type = building_type:cookery
						instant = yes
						cost_multiplier = 0
						cost_multiplier_reason = "game_start"
					}
				}
				else_if = {
					limit = {
						location_rank = location_rank:city
						NOT = { has_building = building_type:cookery }
					}
					construct_building = {
						building_type = building_type:cookery
						instant = yes
						cost_multiplier = 0
						cost_multiplier_reason = "game_start"
					}
				}

				if = {
					limit = {
						OR = {
							AND = {
								location_rank = location_rank:town
								population > 20
							}
							AND = {
								location_rank = location_rank:city
								population > 20
							}
							AND = {
								location_rank = location_rank:rural_settlement
								population > 500
							}
						}
						NOT = { has_building = building_type:victuals_market }
					}
					construct_building = {
						building_type = building_type:victuals_market
						instant = yes
						cost_multiplier = 0
						cost_multiplier_reason = "game_start"
					}
				}

				if = {
					limit = {
						population > 74
						OR = {
							raw_material = goods:wheat
							raw_material = goods:maize
							raw_material = goods:rice
							raw_material = goods:millet
							raw_material = goods:legumes
							raw_material = goods:potato
							raw_material = goods:livestock
							raw_material = goods:olives
							raw_material = goods:fruit
							raw_material = goods:wool
						}
					}
					construct_building = {
						building_type = building_type:farming_village
						instant = yes
						cost_multiplier = 0
						cost_multiplier_reason = "game_start"
					}
				}

				if = {
					limit = {
						population > 74
						OR = {
							raw_material = goods:fish
							raw_material = goods:salt
							raw_material = goods:amber
							raw_material = goods:pearls
						}
					}
					construct_building = {
						building_type = building_type:fishing_village
						instant = yes
						cost_multiplier = 0
						cost_multiplier_reason = "game_start"
					}
				}

				if = {
					limit = {
						population > 74
						raw_material = goods:wool
					}
					construct_building = {
						building_type = building_type:sheep_farms
						instant = yes
						cost_multiplier = 0
						cost_multiplier_reason = "game_start"
					}
				}

				if = {
					limit = {
						population > 74
						OR = {
							raw_material = goods:wild_game
							raw_material = goods:fur
						}
					}
					construct_building = {
						building_type = building_type:forest_village
						instant = yes
						cost_multiplier = 0
						cost_multiplier_reason = "game_start"
					}
				}

				if = {
					limit = {
						population > 74
						raw_material = goods:fruit
					}
					construct_building = {
						building_type = building_type:fruit_orchard
						instant = yes
						cost_multiplier = 0
						cost_multiplier_reason = "game_start"
					}
				}
			}
		}
	}
}
