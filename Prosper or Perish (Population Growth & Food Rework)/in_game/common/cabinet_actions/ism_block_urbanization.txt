# Overlord actions to block/allow subject urbanization.
# A subject with the modifier "ism_overlord_blocking_urbanization" pays massively increased costs for town upgrades.

ism_block_subject_urbanization = {
    # Only show this action when there is at least one subject that can be blocked.
    visible = {
        any_subject = {
            NOT = { has_country_modifier = ism_overlord_blocking_urbanization }
        }
    }

    select_trigger = {
        looking_for_a = subject
        source = actor
        target_flag = target
        name = "ism_block_subject_urbanization_select_subject"
        none_available_msg_key = "ism_block_subject_urbanization_no_subjects"

        visible = {
            scope:target = {
                NOT = { has_country_modifier = ism_overlord_blocking_urbanization }
            }
        }
    }

    effect = {
        scope:target = {
            add_country_modifier = { modifier = ism_overlord_blocking_urbanization }
        }
    }
}

ism_allow_subject_urbanization = {
    # Only show this action when there is at least one subject that currently has the modifier.
    visible = {
        any_subject = {
            has_country_modifier = ism_overlord_blocking_urbanization
        }
    }

    select_trigger = {
        looking_for_a = subject
        source = actor
        target_flag = target
        name = "ism_allow_subject_urbanization_select_subject"
        none_available_msg_key = "ism_allow_subject_urbanization_no_subjects"

        visible = {
            scope:target = {
                has_country_modifier = ism_overlord_blocking_urbanization
            }
        }
    }

    effect = {
        scope:target = {
            remove_country_modifier = { modifier = ism_overlord_blocking_urbanization }
        }
    }
}
