# This file is mostly used to reduce existing population growth modifiers to 0
# Generally the idea is to disentangle pop growth from everything that has nothing to do with food growth (mostly)
# If you provide a lot of food in the province, growth is massively increased, if not provinces grow very little
# This modifier is population growth impact at 100 prosperity (linearly scaled down if lower)
TRY_INJECT:prosperity = {
	local_population_growth = -0.002
	local_monthly_development = 0.001
	local_monthly_development_modifier = 0.25
	local_peasants_food_consumption = 0.10
	local_unrest = -0.03
}
# I think it makes some sense for people to die when looting
# Idea is that food stocks are raided
