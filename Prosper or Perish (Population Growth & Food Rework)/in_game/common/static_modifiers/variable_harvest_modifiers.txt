# General idea is similar to variable harvest, but variable harvests should affect different foods differently
# Then we scale these around 7 different options (where option 4) roughly normal distributed probabilities
# longterm ev should be 0 for all crops. We just want to add variance
# Issue with this is, we dont hit peasant production, because its governed by SUBSISTENCE_AGRICULTURE, so we also need to include local_monthly_food_modifier
# It needs to be seen how much these changes disturb production further down the line, maybe market stockpiles need to be adjusted.

abysmal_harvest_modifier = {
	game_data = { category = location }

	# To hit peasants aswell.
	local_monthly_food_modifier = 0.0

	# Highly affected (-100% Scale)
	local_wheat_output_modifier = -0.75
	local_millet_output_modifier = -0.75
	local_maize_output_modifier = -0.60
	local_fruit_output_modifier = -0.80
	
	# Slightly less variable
	local_rice_output_modifier = -0.60
	local_beeswax_output_modifier = -0.50

	# Very low
	local_legumes_output_modifier = -0.25
	local_potato_output_modifier = -0.15
	local_olives_output_modifier = -0.30
	local_wool_output_modifier = -0.20

	# Nearly Independant
	local_livestock_output_modifier = -0.10
	local_wild_game_output_modifier = -0.10
	local_fish_output_modifier = 0.00
	local_fur_output_modifier = 0.00
}
